    module JSONAPIObjects
  module MemberNames

    def self.extended(obj)
      raise TypeError unless obj.is_a?(String) || mod <= String
    end

    def self.included(obj)
      raise TypeError unless obj <= String
    end

    def valid_member_name?
      [
        present?,
        valid_ends?,
        contains_valid_characters?,
        !contains_invalid_characters?
      ].reduce(:&)
    end

    private

    def contains_valid_characters?
      self =~ /\A[a-zA-Z0-9\u0080-\uFFFF_\s-]+\Z/
    end

    def valid_ends?
      ['-', '_', ' '].map do |char|
        !start_with?(char) & !end_with?(char)
      end.reduce(:&)
    end

    def contains_invalid_characters?
      %w{+ , . [ ] ! " # $ % & ' ( ) * / : ; < = > ? @ \\ ^ ` { } | ~}.map do |char|
        include? char
      end.reduce(:|)
    end

  end
end
