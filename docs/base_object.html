    Fails if these keys exist deeply

          def must_not_contain_deep!(*keys, &block)
        set_callback :compile, :before do
          keys       += self.keys.select(&block) if block_given?
          is_invalid = proc do |hash|
            invalid_keys = hash.keys.map(&:to_sym) & keys.map(&:to_sym)
            children     = hash.values.select { |v| v.respond_to?(:to_hash) }
            invalid_keys.present? | children.map { |c| is_invalid(c) }.reduce(:|)
          end
          if is_invalid.call(self)
            message = "deeply not permitted: #{invalid_keys.join(', ')}."
            raise validation_error message
          end
        end
      end

          def must_contain!(*keys, origin: nil, &block)
        self.permitted_keys = [*permitted_keys, *keys].uniq
        set_callback :compile, :before do
          keys         += self.keys.select(&block) if block_given?
          missing_keys = keys.map(&:to_sym) - self.keys.map(&:to_sym)
          if (origin.nil? || origin == self.origin) && missing_keys.present?
            raise validation_error "missing: #{missing_keys.join(', ')}."

    Validates key using a provided method or block

          def validate!(key = nil, symbol = nil, message: nil, &block)
        raise ArgumentError, 'cannot pass symbol and block.' if symbol && block
        set_callback :compile, :before do
          message ||= "is not valid."
          block   ||= method(symbol).to_proc
          args    = [self]
          args    += [key, self[key]] if key
          unless block.call(*args)
            raise validation_error "`#{key}` #{message}"
          end
        end
      end

          def validate_each!(with: nil, message: nil, &block)
        raise ArgumentError, 'cannot pass symbol and block.' if with && block
        set_callback :compile, :before do
          message ||= "not valid"
          block   ||= method(symbol).to_proc
          keys.each do |key|
            unless block.call(self, key, self[key])
              raise validation_error "`#{key}` #{message}"

          def type_of!(key, must_be:, allow_nil: false, message: nil)
        types   = Array.wrap(must_be)
        message ||= "must be a #{types.map(&:name).to_sentence(last_word_connector: ', or ', two_words_connector: ' or ')}."
        validate! key, message: message do |*, value|
          (value.nil? && allow_nil) || types.any? { |type| value.is_a? type }
        end
      end
    end

    def coerce_value(k, v)
      if implementations[k]
        implementations[k].new(v).tap do |value|
          value.instance_variable_set(:@parent, self)
        end
      else
        v
      end
    end

    def object
      @object.tap do
        newly_set = unset.select { |_, v| v.present? }
        newly_set.each do |k, v|
          self[k] = v
        end
      end
    end
